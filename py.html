# the Count class.  The wordleFromObject function takes a Count object as

class Count:
        # dictionary of word counts
        wordCounts = None

        # method to initialize any data structures, such as a dictionary to
        # hold the counts for each word, and a list of stop words
        def __init__(self):
                print("Initializing Word Counter")
                # set the attrbute wordCounts to an empty dictionary
                self.wordCounts = {}
                self.stopWords = []
                #add stop words into dictionary
                stopFile = open("stopWords.txt")
                for line in stopFile:
                        line = line.strip()
                        self.stopWords.append(line)

        # method to add one to the count for a word in the dictionary.
        # if the word is not yet in the dictionary, we'll need to add a
        # record for the word, with a count of one. 
        def incCount(self,word):
                word = clean(word)
                # IF these strings are in the file then make it 0
                if word in self.stopWords:
                        return 0       
                # Do not add the empty string to the dictionary
                if word == "":
                        return
                # Add code here to update the dictionary
                if word in self.wordCounts:
                        self.wordCounts[word] = self.wordCounts[word]+1
                if word not in self.wordCounts:
                        self.wordCounts[word] = 1
                return

        # method to look up the count for a word
        def lookUpCount(self,word):
                if word not in self.wordCounts:
                        self.wordCounts[word] = 0
                return self.wordCounts[word]

        # method to get the most frequent words out of the dictionary.
        # The argument "num" indicates how many words to return. 
        def getTopWords(self,num):
                turnToList = list(self.wordCounts.items())
                wordCountsList = []
                for key, value in turnToList:
                        wordCountsList.append((value, key))
                wordCountsList.sort (reverse=True)
                finalList = []
                for i in range(num):
                        finalList.append(wordCountsList[i])
                return finalList

class WordRecord:

        self.size = size
        self.vert = vert 

# a position vector that knows how to move in a spiral
class spiralInc:

  def __init__(self,size):
    self.stepSize = size

  def initialize(self,x,y):
    self.x = triangular(x-x/4, x+x/4, x)
    self.y = triangular(y-y/4, y+y/4, y)
    self.initialX = self.x
    self.initialY = self.y


  def update(self,w):
    x = self.x-self.initialX
    y = self.y-self.initialY
    length = math.sqrt(x**2 + y**2)
    # very near the origin
    if length < 1:
      self.x += self.stepSize
      return
    # vector pointing out towards current position, gets smaller linearly with length
    outX = x/(length**2)
    outY = y/(length**2)
    # unit vector perpendicular to vector to current position
    dirX = -y/length
    dirY = x/length
    self.x += self.stepSize*(outX + dirX)
    self.y += self.stepSize*(outY + dirY)


# takes a word object and a place object as input
# adds the word to the wordle at the given place
def addToWordle(word, place, fonts, w):
        fontObj = fonts[word.size]

        else:
            ang = 0
        # actually draw it
        m = w.create_text(place.x, place.y,
              text=word.text, angle=ang, font=fontObj)
        return m

  # move in a spiral trying to find a free location

def montyalways():
    prize = random.randint(1,3) # random # of the door that has a prize in it
    door1 = random.randint(1,3) # random # of the door that contestants select
    if prize == door1:
        return False
    if prize != door1:
        return True

def montynever():
    prize = random.randint(1,3)
    door1 = random.randint(1,3)
    if prize == door1:
        return True
    if prize != door1:
        return False

def montyrandom():
    mr = random.randint(1,2)
    if mr == 1:
        n1 = montyalways()
        return n1
    else:
        n1 = montynever()
        return n1

    class Node:

        # create it
        def __init__(self, data, left= None, right = None):
            self.data = data
            self.count = 1
            self.left = left
            self.right = right

        # generate a string form
        def __str__(self):
            return str(self.data) +" "+ "(" + str(self.count) + ")"

    # A function to insert a new node with the given word
    #
    def insert(root, word):
        # if the tree is empty, return a new node
        if not root:
            return Node(word)
        # if word is the same -> add 1 to the counter
        elif root.data == word:
            root.count += 1
        # Add two blank spaces in front of the line in every recursion
        #
        # if word is lesser than root word -> put it left
        elif word < root.data:
            root.left = insert(root.left, "  " + str(word))
        # if word is greater than root word -> put it right
        else:
            root.right = insert(root.right, "  " + str(word))
        return root
